Pregunta: ¿Cuál de las siguientes opciones es correcta?

  • Respuesta 2

  • Respuesta 4
document.addEventListener("DOMContentLoaded", function () { const opciones = document.querySelectorAll("#opciones li"); const imagen = document.getElementById("imagen"); opciones.forEach(function (opcion) { opcion.addEventListener("click", function () { const respuesta = opcion.getAttribute("data-respuesta"); mostrarImagen(respuesta); }); }); function mostrarImagen(respuesta) { if (respuesta === "opcion1") { imagen.src = "resena.png"; } else if (respuesta === "opcion2") { imagen.src = "imagen2.jpg"; } else if (respuesta === "opcion3") { imagen.src = "imagen3.jpg"; } else if (respuesta === "opcion4") { imagen.src = "imagen4.jpg"; } imagen.style.display = "block"; } });
